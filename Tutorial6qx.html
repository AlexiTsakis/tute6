x <- seq(1:100)
y <- rnorm(100) # Generating data form a normal distribution with mean 0 and variance 1
data <- data.frame(x = x, y = y ) # Creating a data frame
p1 = ggplot(data, aes(x= x, y = y)) +
geom_line()
p2 = ggplot(data, aes(x= x, y = y)) +
geom_point()
